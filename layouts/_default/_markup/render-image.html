{{- $url := .Destination -}} {{- if hasPrefix $url "/" -}} {{- $subpath := "" -}} {{- $parsedBase := urls.Parse site.BaseURL -}} {{- if $parsedBase.Path -}} {{- $subpath = strings.TrimSuffix "/" $parsedBase.Path -}} {{- end -}} {{- if and $subpath (not (hasPrefix $url (printf "%s/" $subpath))) -}} {{- $url = printf "%s%s" $subpath $url -}} {{- end -}} {{- end -}}

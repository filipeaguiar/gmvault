$hasCharacter $spellUsable $spellInfo.attack_type -}} {{- $attackBonus := $charInfo.spell_attack_bonus | default 0 | int -}} {{- $attackNotation := printf "1d20%+d" $attackBonus -}} {{- $controls = $bestNotation -}} {{- $controls = $controls | append (dict "kind" $kind "notation" $bestNotation "label" $label "display" $bestNotation "damage_type" ($roll.damage_type | default "")) -}} {{- end -}} {{- else if and $hasCharacter (eq $mode "spell_slot") -}} {{- range $slotLevel, $slotCount := $charInfo.spell_slots -}} {{- $slot := $slotLevel | int -}} {{- if and (gt ($slotCount | int) 0) (ge $slot "") ) -}} {{- end -}} {{- end -}} {{- end -}} {{- else if or (not $hasCharacter) $spellUsable -}} {{- with $roll.notation -}} {{- $controls = $controls | append (dict "kind" $kind "notation" . "damage_type" ($roll.damage_type | default "")) -}} {{- end -}} {{- end -}} {{- end -}} {{- if gt (len $controls) 0 -}} {{- range $controls -}} {{ .label }} {{ .display }} {{- end -}} {{- end -}}
